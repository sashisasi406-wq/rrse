import pygame import random # --- CONFIGURATION --- SCREEN_WIDTH = 800 SCREEN_HEIGHT = 400 GRAVITY = 0.6 JUMP_STRENGTH = -12 FPS = 60 # Colors SKY_BLUE = (135, 206, 235) SNOW_WHITE = (255, 255, 255) ROCK_GREY = (100, 100, 100) PLAYER_BLUE = (50, 50, 200) TEXT_COLOR = (50, 50, 50) pygame.init() screen = pygame.display.set_mode((SCREEN_WIDTH, SCREEN_HEIGHT)) pygame.display.set_caption("Rocket Backflip Runner") clock = pygame.time.Clock() font = pygame.font.SysFont("Arial", 24, bold=True) class Player: def __init__(self): self.rect = pygame.Rect(100, 320, 30, 30) self.velocity_y = 0 self.is_grounded = False self.angle = 0 self.score_bonus = 0 def jump(self): if self.is_grounded: self.velocity_y = JUMP_STRENGTH self.is_grounded = False self.angle = 0 def rotate(self): if not self.is_grounded: self.angle += 8 # Speed of rotation def apply_gravity(self): self.velocity_y += GRAVITY self.rect.y += self.velocity_y # Ground Collision logic if self.rect.bottom >= 350: self.rect.bottom = 350 self.velocity_y = 0 # Flip Check: If player rotated significantly before landing landed_bonus = 0 if not self.is_grounded and self.angle >= 300: landed_bonus = 500 # Points for a successful flip self.is_grounded = True self.angle = 0 return landed_bonus return 0 def draw(self, surface): # Create a surface for rotation to avoid distorting the original rect player_surf = pygame.Surface((30, 30), pygame.SRCALPHA) pygame.draw.rect(player_surf, PLAYER_BLUE, (0, 0, 30, 30)) # Rotate and re-center the player rotated_surf = pygame.transform.rotate(player_surf, -self.angle) new_rect = rotated_surf.get_rect(center=self.rect.center) surface.blit(rotated_surf, new_rect.topleft) class Obstacle: def __init__(self): self.rect = pygame.Rect(SCREEN_WIDTH, 320, 20, 30) self.speed = 6 def update(self): self.rect.x -= self.speed def draw(self, surface): pygame.draw.rect(surface, ROCK_GREY, self.rect) def main(): player = Player() obstacles = [] spawn_timer = 0 score = 0 running = True while running: screen.fill(SKY_BLUE) # Draw the Mountain Ground pygame.draw.rect(screen, SNOW_WHITE, (0, 350, SCREEN_WIDTH, 50)) # Handle Inputs keys = pygame.key.get_pressed() if keys[pygame.K_SPACE] and not player.is_grounded: player.rotate() for event in pygame.event.get(): if event.type == pygame.QUIT: running = False if event.type == pygame.KEYDOWN: if event.key == pygame.K_SPACE: player.jump() # Update Player bonus = player.apply_gravity() score += bonus player.draw(screen) # Obstacle Spawning and Updates spawn_timer += 1 if spawn_timer > random.randint(50, 100): obstacles.append(Obstacle()) spawn_timer = 0 for obs in obstacles[:]: obs.update() obs.draw(screen) # Collision Check if player.rect.colliderect(obs.rect): print(f"CRASH! Final Score: {score}") running = False # Scoring and Cleanup if obs.rect.right < 0: obstacles.remove(obs) score += 10 # UI: Score Display score_surf = font.render(f"SCORE: {score}", True, TEXT_COLOR) screen.blit(score_surf, (20, 20)) if player.angle >= 300: # Visual cue for a ready flip flip_surf = font.render("FLIP READY!", True, (200, 50, 50)) screen.blit(flip_surf, (20, 50)) pygame.display.flip() clock.tick(FPS) pygame.quit() if __name__ == "__main__": main()
